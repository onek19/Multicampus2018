### 테이블 복사

 ~~~SQL
 create table [테이블명]
   as
   select [필드명] ... from [테이블명];
~~~

<예제> deptno가 20인 사람들의 이름, 직업, 부서번호로 이루어진 테이블 복사
~~~SQL
create table c_emp_20
  as select ename, job, deptno from emp where deptno = 20;
~~~

~~~SQL
ename     job     deptno
========================
SMITH	CLERK  	20
JONES	MANAGER	20
SCOTT	ANALYST	20
ADAMS	CLERK	  20
FORD 	ANALYST	20
~~~
